import paramiko

# assume `private_key_file` is the path to the private key file
try:
    # read the private key file using paramiko
    private_key = paramiko.RSAKey.from_private_key_file(private_key_file)
except paramiko.ssh_exception.SSHException as e:
    # handle any errors that occur during reading
    print(f"Error reading private key: {e}")
else:
    # convert the private key to PuTTY format
    putty_key = private_key.get_putty_private_key()
    
    # write the converted key to a file
    with open("putty_key.ppk", "w") as f:
        f.write(putty_key)
